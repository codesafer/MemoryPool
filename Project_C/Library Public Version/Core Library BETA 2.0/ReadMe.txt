사용 방법
	초기화를 처리할 cpp 파일에서 다음 파일을 포함(include)
		./User/Core_import.h
	초기화
		::CORE::Connect_CORE(...)
	종료(명시적 종료 시점이 필요할때만 사용)
		::CORE::Disconnect_CORE

	사용을 위해서는 다음 파일을 포함(include)
		./User/Core_include.h
//----------------------------------------------------------------
인터페이스는 다음 파일들을 참조
	Core_Interface.h
	MemoryPool_Interface.h
	System_Information_Interface.h

	LogWriter_Interface.h
